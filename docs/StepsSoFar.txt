🐍 The Code (main.py)
Open main.py in your preferred editor (or via nano main.py) and paste this boilerplate. This is a "Production-Ready" starting point that handles the Meta verification challenge and the Firestore write.

Python
import functions_framework
from google.cloud import firestore
import json

# Initialize Firestore Client
db = firestore.Client()

@functions_framework.http
def meta_webhook(request):
    # 1. Handle Meta Webhook Verification (The "Handshake")
    if request.method == 'GET':
        mode = request.args.get('hub.mode')
        token = request.args.get('hub.verify_token')
        challenge = request.args.get('hub.challenge')
        
        # Replace 'YOUR_VERIFY_TOKEN' with a secure string later
        if mode == 'subscribe' and token == 'elite_build_2026':
            return challenge, 200
        return 'Verification failed', 403

    # 2. Handle Incoming Lead Data
    if request.method == 'POST':
        data = request.get_json()
        
        # Log for debugging (viewable in GCP Logs)
        print(f"Received lead: {json.dumps(data)}")
        
        # Extract basic info (Meta payload structure varies by API version)
        # For MVP, we save the raw payload to ensure no data loss
        doc_ref = db.collection('leads').document()
        doc_ref.set({
            'raw_data': data,
            'status': 'New',
            'source': 'Meta Ads',
            'created_at': firestore.SERVER_TIMESTAMP
        })
        
        return 'Lead Captured', 200

import functions_framework
from google.cloud import firestore

db = firestore.Client()

@functions_framework.cloud_event
def match_lead_to_inventory(cloud_event):
    # 1. Get the new Lead data
    data = cloud_event.data
    event_id = cloud_event["id"]
    resource_name = cloud_event["source"]
    
    lead_data = data["value"]["fields"]
    # Adjusting for Firestore's internal JSON format
    interest = lead_data.get("interest", {}).get("stringValue", "Unknown")
    lead_id = resource_name.split('/')[-1]

    print(f"🔎 Processing Lead {lead_id} interested in {interest}")

    # 2. Query Inventory for matches
    # Find plots that match the size AND are Available
    inventory_query = db.collection('inventory') \
        .where('size', '==', interest) \
        .where('status', '==', 'Available') \
        .limit(1).get()

    if inventory_query:
        matched_plot = inventory_query[0].id
        print(f"✅ Match Found: {matched_plot}")
        
        # 3. Update the Lead with the suggested plot
        db.collection('leads').document(lead_id).update({
            'suggested_plot': matched_plot,
            'status': 'Matched',
            'matched_at': firestore.SERVER_TIMESTAMP
        })
    else:
        print(f"❌ No available {interest} plots found.")

import functions_framework
from google.cloud import firestore

db = firestore.Client()

@functions_framework.cloud_event
def match_lead_to_inventory(cloud_event):
    data = cloud_event.data
    lead_data = data["value"]["fields"]
    
    # Extracting the new critical attributes
    budget = float(lead_data.get("budget", {}).get("doubleValue", 0))
    pref_facings = [f.get("stringValue") for f in lead_data.get("pref_facings", {}).get("arrayValue", {}).get("values", [])]

To enable Auto Save in VSCode, you have two quick options:

1. The Quick Menu Method (Fastest)
Go to the top menu bar.

Click on File.

Click on Auto Save. (A checkmark will appear next to it once enabled).

2. The Settings Method (More Control)
If you want to choose when it saves (e.g., as soon as you stop typing vs. when you click away), do this:

Press Cmd + , (Mac) or Ctrl + , (Windows) to open Settings.

Search for "Auto Save" in the top search bar.

Locate the Files: Auto Save dropdown. You have four choices:

off: Manual saving only.

afterDelay: Saves after a specific amount of time (set the delay in the "Auto Save Delay" setting below it—default is 1000ms).

onFocusChange: Saves when you click out of the current editor tab.

onWindowChange: Saves when you switch to a different application (like your terminal or browser).

💡 Senior DevOps Tip:
For your Elite Build CRM development, I recommend setting it to onFocusChange. This ensures your files are always saved before you run a gcloud deployment command in the terminal, preventing you from accidentally deploying an old version of your code.
